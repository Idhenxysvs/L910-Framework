# Веб сайт расписания концертов - Lab 9-10

Данный веб сайт имеет функционал создания концертов и артистов, а также расписание концертов со всей нужной информацией.
Веб сайт представлен панелью управлением концертами и артистами. Панель позволяяет настраивать концерты, назначать на них артиста, а также вносить самого артиста и данные о его деятельности.

## Структура объектов сайта:  
Объект "Артист":  
{  
&nbsp;&nbsp;"id": 1, – уникальный идентификатор артиста  
&nbsp;&nbsp;"name": "имя_артиста", — имя или псевдоним артиста  
&nbsp;&nbsp;"country": "название_страны", — страна артиста  
&nbsp;&nbsp;"active": true/false, — активность артиста в настоящий момент  
&nbsp;&nbsp;"albumCount": 0, — количетсов альбовом вшедшие в релиз  
&nbsp;&nbsp;"popularity": 0, — рейтинг артиста по 10-бальной оценке  
&nbsp;&nbsp;"upcomingConcerts": [], — ID концертов, на которых выступает артист  
&nbsp;&nbsp;"debutDate": "2025-12-31", — первое выступление артиста  
&nbsp;&nbsp;"createdAt": "2025-12-26T09:44:59.081Z" – дата создания объекта  
}  
Объект "Концерт":  
{  
&nbsp;&nbsp;"id": 1, — уникальный идентификатор концерта  
&nbsp;&nbsp;"title": "название_концерта", — название концерта  
&nbsp;&nbsp;"artistId": 1, — ID артиста, выступающий на концерте  
&nbsp;&nbsp;"date": "2025-12-31T17:00", — дата и время проведения концерта  
&nbsp;&nbsp;"venue": "место_проведения", — адрес/место проведения концерта  
&nbsp;&nbsp;"price": 10, — стоимость билета  
&nbsp;&nbsp;"soldOut": false, — распроданы ли все билеты на концерт  
&nbsp;&nbsp;"genres": ["Жанр1", "Жанр2", "Жанр3"], — ключевые жанры концерта  
&nbsp;&nbsp;"createdAt": "2025-12-22T12:00:00.000Z", — дата и время создания концерта  
&nbsp;&nbsp;"updatedAt": "2025-12-25T18:34:20.640Z", — дата и время обновления данных концерта  
}  

## API Документация:

GET    /concerts 		— список всех концертов;
GET    /concerts/:id 	— концерт по ID;
POST   /concerts 		— создать новый концерт;
PUT    /concerts/:id 	— обновить концерт;
PATCH  /concerts/:id 	— частично обновить концерт;
DELETE /concerts/:id 	— удалить концерт.

GET    /artists 		— список всех артистов;
GET    /artists/:id 	— артист по ID;
POST   /artists 		— создать нового артиста;
PUT    /artists/:id 	— обновить артиста;
PATCH  /artists/:id 	— частично обновить артиста;
DELETE /artists/:id 	— удалить артиста.
